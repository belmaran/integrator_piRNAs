pairwise_fc<-function(df,col1,col2,pseudoTPMs){
  df<-df[which((df[,col1]+df[,col2])>0),]
  return(log2(df[,col2]+pseudoTPMs)-log2(df[,col1]+pseudoTPMs))
}
